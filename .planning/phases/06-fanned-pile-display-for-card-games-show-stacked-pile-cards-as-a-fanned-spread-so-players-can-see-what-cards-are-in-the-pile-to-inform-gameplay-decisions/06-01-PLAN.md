  if (pile.length === 0) return 0;
  const topRank = pile[pile.length - 1].rank;
  let count = 0;
  for (let i = pile.length - 1; i >= 0; i--) {
    if (pile[i].rank === topRank) count++;
    else break;
  }
  return count;
}

```js
function renderFannedPile(pileCards, opts) {
  opts = opts || {};
  var cardWidth = opts.cardWidth || 72;
  var cardHeight = opts.cardHeight || 100;
  var stepPx = opts.stepPx || 20;
  var maxCards = 5;

  if (!pileCards || pileCards.length === 0) {
    return '<div style="display:inline-flex;align-items:center;justify-content:center;' +
      'width:' + cardWidth + 'px;height:' + cardHeight + 'px;' +
      'border-radius:12px;border:3px dashed rgba(255,255,255,0.15);' +
      'color:var(--text-secondary);font-size:1.2rem;">Empty</div>';
  }

  var containerHeight = (maxCards - 1) * stepPx + cardHeight;
  var html = '<div class="pile-fan" style="position:relative;width:' + cardWidth +
    'px;height:' + containerHeight + 'px;display:inline-block;">';

  var count = pileCards.length;
  var topIndex = count - 1;
  for (var i = 0; i < count; i++) {
    var offsetTop = (topIndex - i) * stepPx;
    var zIndex = i + 1;
    var isTop = (i === topIndex);
    var extraClass = isTop ? '' : 'fan-card';
    html += renderCardImg(pileCards[i], {
      width: cardWidth,
      height: cardHeight,
      className: extraClass,
      style: 'position:absolute;left:0;top:' + offsetTop + 'px;z-index:' + zIndex
    });
  }

  html += '</div>';
  return html;
}
```

Key behaviors per UI-SPEC.md:
- Container is position:relative, fixed height based on maxCards (not actual count) so layout doesn't shift
- Cards positioned absolutely, top card at top:0 with highest z-index
- Each lower card offset downward by stepPx
- Non-top cards get 'fan-card' class for shadow
- Empty state renders dashed border placeholder
- Caller passes cardWidth/cardHeight/stepPx for TV vs phone sizing
